        // Simple Interaction Logic
        document.querySelectorAll('aside nav button').forEach(button => {
            button.addEventListener('click', function() {
                // Remove active classes from all
                document.querySelectorAll('aside nav button').forEach(b => {
                    b.classList.remove('text-primary', 'font-bold');
                    b.classList.add('text-secondary');
                    const indicator = b.querySelector('.active-indicator');
                    if (indicator) indicator.remove();
                });
                
                // Add to clicked
                this.classList.add('text-primary', 'font-bold');
                this.classList.remove('text-secondary');
                const newIndicator = document.createElement('div');
                newIndicator.className = 'active-indicator';
                this.appendChild(newIndicator);
            });
        });

        // Search focus interaction
        const searchInput = document.querySelector('input[type="text"]');
        searchInput.addEventListener('focus', () => {
            searchInput.parentElement.classList.add('scale-[1.02]');
        });
        searchInput.addEventListener('blur', () => {
            searchInput.parentElement.classList.remove('scale-[1.02]');
        });

        function switchTab(tabId, el) {
            // Update Tab Panes
            document.querySelectorAll('.tab-pane').forEach(pane => {
                pane.classList.add('hidden');
                pane.classList.remove('block');
            });
            document.getElementById(tabId).classList.remove('hidden');
            document.getElementById(tabId).classList.add('block');

            // Update Tab Button Styles
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('text-primary', 'font-bold');
                btn.classList.add('text-secondary');
            });
            el.classList.add('text-primary', 'font-bold');
            el.classList.remove('text-secondary');

            // Move Indicator
            const indicator = document.getElementById('active-indicator');
            const rect = el.getBoundingClientRect();
            const parentRect = el.parentElement.getBoundingClientRect();
            
            indicator.style.width = el.offsetWidth + 'px';
            indicator.style.left = (rect.left - parentRect.left) + 'px';
        }
